// ─── UTILITIES ─────────────────────────────────────────────────────────────────
function formatDate(d) {
  if (!d) return "";
  return new Date(d).toLocaleDateString("en-US",{month:"short",day:"numeric",year:"numeric"});
}
function pct(a,b) { return b ? Math.round((a/b)*100) : 0; }

// ─── TOAST CONTEXT ─────────────────────────────────────────────────────────────
function useToast() {
  const [toasts, setToasts] = useState([]);
  const add = useCallback((msg, type="success") => {
    const id = Date.now();
    setToasts(t => [...t, {id, msg, type}]);
    setTimeout(() => setToasts(t => t.filter(x => x.id !== id)), 3500);
  }, []);
  return { toasts, add };
}

function ToastContainer({ toasts }) {
  return (
    <div className="toast-container">
      {toasts.map(t => (
        <div key={t.id} className={`toast ${t.type}`}>
          {t.type === "success" ? "✅" : t.type === "error" ? "❌" : "ℹ️"} {t.msg}
        </div>
      ))}
    </div>
  );
}

// ─── MODAL ─────────────────────────────────────────────────────────────────────
function Modal({ title, onClose, children, maxWidth=520 }) {
  useEffect(() => {
    const fn = e => e.key === "Escape" && onClose();
    window.addEventListener("keydown", fn);
    return () => window.removeEventListener("keydown", fn);
  }, [onClose]);
  return (
    <div className="modal-overlay" onClick={e => e.target === e.currentTarget && onClose()}>
      <div className="modal" style={{maxWidth}}>
        <div className="flex items-center justify-between mb-16">
          <div className="modal-title">{title}</div>
          <button onClick={onClose} style={{background:"none",border:"none",fontSize:20,cursor:"pointer",color:"var(--text-muted)"}}>✕</button>
        </div>
        {children}
      </div>
    </div>
  );
}

// SCREEN 4 — ITINERARY BUILDER
// ══════════════════════════════════════════════════════════════════════════════
function ItineraryBuilder({ trip, setScreen, toast }) {

  const addSection = () => setSections(s => [...s, {
    id: `s${Date.now()}`, title:`Section ${s.length+1}`, description:"", startDate:"", endDate:"", budget:0
  }]);
  const delSection = id => setSections(s => s.filter(x => x.id !== id));
  const upd = (id, k, v) => setSections(s => s.map(x => x.id===id ? {...x,[k]:v} : x));

  const totalBudget = sections.reduce((a,s)=>a+(Number(s.budget)||0),0);

  const save = async () => {
    await new Promise(r=>setTimeout(r,800));
    toast("Itinerary saved! 🗺", "success");
  };

  return (
    <div className="page">
      <div className="page-wide">

          {trip && <span className="pill pill-info">{trip.name}</span>}
        </div>
        <div style={{display:"grid",gridTemplateColumns:"1fr 280px",gap:24}}>
          <div className="flex-col gap-16">
            {sections.map((sec,idx) => (
              <div key={sec.id} className="section-card">
                <div className="flex items-center justify-between mb-12">
                  <div className="flex items-center gap-10">
                    <span className="drag-handle">⠿</span>

                  </div>
                  <button style={{background:"none",border:"none",cursor:"pointer",fontSize:18,color:"var(--danger)"}} onClick={()=>delSection(sec.id)}>🗑</button>
                </div>
                <div className="input-group mb-12">
                  <label className="input-label">Description</label>
                  <textarea className="input textarea" style={{minHeight:70}} placeholder="All necessary info — travel, hotel, activity..." value={sec.description} onChange={e=>upd(sec.id,"description",e.target.value)} />
                </div>
                <div className="grid-2 gap-12 mb-12">
                  <div className="input-group">
                    <label className="input-label">Start Date</label>
                    <input type="date" className="input" value={sec.startDate} onChange={e=>upd(sec.id,"startDate",e.target.value)} />
                  </div>
                  <div className="input-group">
                    <label className="input-label">End Date</label>
                    <input type="date" className="input" value={sec.endDate} onChange={e=>upd(sec.id,"endDate",e.target.value)} />
                  </div>
                </div>
                <div className="input-group">
                  <label className="input-label">Budget for this Section ($)</label>
                  <input type="number" className="input" value={sec.budget} onChange={e=>upd(sec.id,"budget",e.target.value)} />

            <button className="btn btn-ghost w-full" style={{borderStyle:"dashed",borderWidth:2,padding:16}} onClick={addSection}>

            </button>

              <button className="btn btn-primary" onClick={save}>Save Itinerary</button>
            </div>
          </div>

          {/* Sidebar */}
          <div className="flex-col gap-16">
            <div className="budget-panel">
              <h3 style={{marginBottom:16}}>Budget Summary</h3>
              {sections.map(s => (

              </div>
            </div>
          </div>
        </div>
      </div>

  );

// SCREEN 10 — PACKING CHECKLIST
// ══════════════════════════════════════════════════════════════════════════════
function ChecklistScreen({ toast }) {
  const [items, setItems] = useState(MOCK.checklist);
  const [newItem, setNewItem] = useState("");
  const [newCat, setNewCat] = useState("Documents");
  const [addModal, setAddModal] = useState(false);
  const [resetModal, setResetModal] = useState(false);

  const toggle = id => setItems(prev => prev.map(x => x.id===id?{...x,packed:!x.packed}:x));
  const del = id => setItems(prev => prev.filter(x=>x.id!==id));
  const addItem = () => {
    if (!newItem) return;
    setItems(prev=>[...prev,{id:`c${Date.now()}`,name:newItem,category:newCat,packed:false}]);
    setNewItem(""); setAddModal(false);
    toast("Item added to checklist!");
  };
  const resetAll = () => { setItems(prev=>prev.map(x=>({...x,packed:false}))); setResetModal(false); toast("Checklist reset"); };

  const cats = [...new Set(items.map(x=>x.category))];
  const totalPacked = items.filter(x=>x.packed).length;

  return (
    <div className="page">
      <div className="page-narrow">
        <div className="flex items-center justify-between mb-20">
          <h1 style={{fontSize:28}}>Packing Checklist</h1>
          <select className="input" style={{width:220}}>
            {MOCK.trips.map(t=><option key={t.id}>{t.name}</option>)}
          </select>
        </div>

        {/* Progress */}
        <div className="card mb-24">
          <div className="flex justify-between mb-8">
            <span className="font-bold">{totalPacked}/{items.length} items packed</span>
            <span className="text-muted text-sm">{pct(totalPacked,items.length)}%</span>
          </div>
          <div className="progress-bar" style={{height:12}}>
            <div className="progress-fill" style={{width:`${pct(totalPacked,items.length)}%`}}/>
          </div>
        </div>

        {/* Categories */}
        {cats.map(cat => {
          const catItems = items.filter(x=>x.category===cat);
          const catPacked = catItems.filter(x=>x.packed).length;
          return (
            <div key={cat} className="card mb-16">
              <div className="flex items-center justify-between mb-12">
                <h3 style={{fontSize:15}}>{cat}</h3>
                <span className="pill pill-info" style={{fontSize:12}}>{catPacked}/{catItems.length}</span>
              </div>
              {catItems.map(item => (
                <div key={item.id} className="checklist-item">
                  <div className={`checkbox ${item.packed?"checked":""}`} onClick={()=>toggle(item.id)}>
                    {item.packed && "✓"}
                  </div>
                  <span className={`item-name ${item.packed?"crossed":""}`} style={{flex:1}}>{item.name}</span>
                  <button style={{background:"none",border:"none",cursor:"pointer",color:"var(--text-muted)",opacity:0,transition:"var(--transition)"}} className="del-btn" onClick={()=>del(item.id)}>🗑</button>
                </div>
              ))}
              <div style={{marginTop:8}}>
                <input className="input" style={{fontSize:13}} placeholder="+ Add item..." onKeyDown={e=>{if(e.key==="Enter"&&e.target.value){setItems(prev=>[...prev,{id:`c${Date.now()}`,name:e.target.value,category:cat,packed:false}]);e.target.value="";}}} />
              </div>
            </div>
          );
        })}

        <div className="flex gap-12">
          <button className="btn btn-primary" onClick={()=>setAddModal(true)}>+ Add Item</button>
          <button className="btn btn-ghost" onClick={()=>setResetModal(true)}>Reset All</button>
          <button className="btn btn-ghost">Share Checklist</button>
        </div>
      </div>

      {addModal && (
        <Modal title="Add Checklist Item" onClose={()=>setAddModal(false)}>
          <div className="input-group mb-12">
            <label className="input-label">Item Name</label>
            <input className="input" value={newItem} onChange={e=>setNewItem(e.target.value)} placeholder="e.g. Sunscreen" autoFocus />
          </div>
          <div className="input-group mb-20">
            <label className="input-label">Category</label>
            <select className="input" value={newCat} onChange={e=>setNewCat(e.target.value)}>
              {["Documents","Clothing","Electronics","Toiletries","Other"].map(c=><option key={c}>{c}</option>)}
            </select>
          </div>
          <div className="flex gap-12">
            <button className="btn btn-ghost flex-1" onClick={()=>setAddModal(false)}>Cancel</button>
            <button className="btn btn-primary flex-1" onClick={addItem}>Add Item</button>
          </div>
        </Modal>
      )}

      {resetModal && (
        <Modal title="Reset Checklist?" onClose={()=>setResetModal(false)}>
          <p className="text-muted mb-24">All items will be marked as unpacked.</p>
          <div className="flex gap-12">
            <button className="btn btn-ghost flex-1" onClick={()=>setResetModal(false)}>Cancel</button>
            <button className="btn btn-danger flex-1" onClick={resetAll}>Reset All</button>
          </div>
        </Modal>
      )}
    </div>
  );
}

function NotesScreen({ toast }) {
  const [notes, setNotes] = useState(MOCK.notes);
  const [q, setQ] = useState("");
  const [groupBy, setGroupBy] = useState("all");
  const [addModal, setAddModal] = useState(false);
  const [newNote, setNewNote] = useState({ title:"", content:"", day:1, stop:"" });

  const filtered = notes.filter(n =>
    (!q || n.title.toLowerCase().includes(q.toLowerCase()) || n.content.toLowerCase().includes(q.toLowerCase()))
  ).sort((a,b) => (b.pinned?1:0)-(a.pinned?1:0));

  const togglePin = id => setNotes(prev => prev.map(n=>n.id===id?{...n,pinned:!n.pinned}:n));
  const delNote  = id => { setNotes(prev=>prev.filter(n=>n.id!==id)); toast("Note deleted"); };
  const saveNote = () => {
    if (!newNote.title) return;
    setNotes(prev=>[{id:`n${Date.now()}`,...newNote,date:new Date().toISOString().slice(0,10),pinned:false},...prev]);
    setNewNote({title:"",content:"",day:1,stop:""});
    setAddModal(false);
    toast("Note saved! 📝");
  };

  const grouped = groupBy === "day"
    ? [...new Set(filtered.map(n=>n.day))].sort().reduce((acc,d)=>({...acc,[d]:filtered.filter(n=>n.day===d)}),{})
    : { all: filtered };

  return (
    <div className="page">
      <div className="page-wide">
        {/* Top bar */}
        <div className="flex gap-12 mb-20 items-center">

function InvoiceScreen({ trip, setScreen, toast }) {
  const inv = MOCK.invoice;

  const [payModal, setPayModal] = useState(false);

  const markPaid = () => { setStatus("paid"); setPayModal(false); toast("Invoice marked as paid! ✅"); };

  return (
    <div className="page">
      <div className="page-wide">
        {/* Nav */}
        <div className="flex items-center justify-between mb-20">
          <div>
            <button className="btn btn-ghost btn-sm mb-8" onClick={()=>setScreen("itinerary-view")}>← Back to My Trips</button>
            <h1 style={{fontSize:26}}>Invoice</h1>
          </div>
          <div className="flex gap-12">
            <input className="input" placeholder="Search invoices..." style={{width:260}} />
            <button className="btn btn-ghost btn-sm">⚙ Filter</button>
            <button className="btn btn-ghost btn-sm">⬆⬇ Sort</button>
          </div>
        </div>

        {/* Invoice Header Card */}
        <div className="card mb-24">
          <div className="invoice-header">
            {/* Left: Trip Info */}
            <div className="flex gap-16 items-start">
              <div style={{width:80,height:80,borderRadius:10,background:"linear-gradient(135deg,var(--primary),var(--accent))",display:"flex",alignItems:"center",justifyContent:"center",fontSize:32,flexShrink:0}}>

              <div className="mb-8"><span className="text-muted text-sm">Generated: </span><span className="text-sm">{formatDate(inv.generatedDate)}</span></div>

                <div className="text-muted text-sm mb-4">Travelers:</div>

            <tbody>
              {inv.lineItems.map((li,i)=>(

                  <td className="mono text-sm">{i+1}</td>
                  <td><span className="pill pill-info" style={{fontSize:12}}>{li.cat}</span></td>
                  <td>{li.desc}</td>

                  <td className="mono">${li.unit.toLocaleString()}</td>

          </table>

            <button className="btn btn-ghost btn-sm mt-12">+ Add Row</button>

          {/* Totals */}
          <div style={{maxWidth:280,marginLeft:"auto",marginTop:20}}>

              <span className="mono" style={{color:"var(--primary)"}}>${inv.grandTotal.toLocaleString()}</span>
            </div>

        <div className="flex gap-12 items-center">

          <button className="btn btn-ghost" onClick={()=>toast("Exporting as PDF...")}>📄 Export as PDF</button>

            <button className="btn btn-primary" style={{marginLeft:"auto"}} onClick={()=>setPayModal(true)}>✅ Mark as Paid</button>
          )}
          {status === "paid" && (
            <span className="pill pill-success" style={{marginLeft:"auto",fontSize:14,padding:"10px 20px"}}>✅ Invoice Paid</span>
          )}
        </div>
      </div>

      {payModal && (
        <Modal title="Mark Invoice as Paid?" onClose={()=>setPayModal(false)}>
          <p className="text-muted mb-8">Invoice: <span className="mono">{inv.number}</span></p>
          <p className="text-muted mb-8">Amount: <span className="font-bold">${inv.grandTotal.toLocaleString()}</span></p>
          <p className="text-muted mb-24">This action will lock the invoice as read-only.</p>
          <div className="flex gap-12">
            <button className="btn btn-ghost flex-1" onClick={()=>setPayModal(false)}>Cancel</button>
            <button className="btn btn-primary flex-1" onClick={markPaid}>Confirm Payment</button>
          </div>
        </Modal>
      )}
    </div>
  );
}

// ROOT APP
// ══════════════════════════════════════════════════════════════════════════════
export default function App() {
  const [user, setUser] = useState(null);
  const [screen, setScreen] = useState("auth");
  const [selectedTrip, setSelectedTrip] = useState(null);
  const { toasts, add: toast } = useToast();

  const handleLogin = (u) => { setUser(u); setScreen("dashboard"); };
  const handleLogout = () => { setUser(null); setScreen("auth"); };

  const renderScreen = () => {
    if (!user) return <AuthScreen onLogin={handleLogin} toast={toast} />;
    switch(screen) {
      case "dashboard":        return <Dashboard user={user} setScreen={setScreen} setSelectedTrip={setSelectedTrip} toast={toast} />;
      case "create-trip":      return <CreateTrip setScreen={setScreen} toast={toast} />;
      case "itinerary-builder":return <ItineraryBuilder trip={selectedTrip} setScreen={setScreen} toast={toast} />;
      case "trips":            return <TripsList setScreen={setScreen} setSelectedTrip={setSelectedTrip} toast={toast} />;
      case "profile":          return <ProfileScreen user={user} setScreen={setScreen} toast={toast} />;
      case "search":           return <SearchScreen setScreen={setScreen} toast={toast} />;
      case "itinerary-view":   return <ItineraryView trip={selectedTrip} setScreen={setScreen} toast={toast} />;
      case "community":        return <CommunityScreen toast={toast} />;
      case "checklist":        return <ChecklistScreen toast={toast} />;
      case "public-trip":      return <PublicTripScreen trip={selectedTrip} setScreen={setScreen} toast={toast} />;
      case "notes":            return <NotesScreen toast={toast} />;
      case "admin":            return <AdminScreen />;
      case "invoice":          return <InvoiceScreen trip={selectedTrip} setScreen={setScreen} toast={toast} />;
      default:                 return <Dashboard user={user} setScreen={setScreen} setSelectedTrip={setSelectedTrip} toast={toast} />;
    }
  };

  // Sidebar screens
  const withSidebar = user && !["auth"].includes(screen);

  return (
    <>
      <style>{CSS}</style>
      <ToastContainer toasts={toasts} />
      {user && <Nav screen={screen} setScreen={setScreen} user={user} onLogout={handleLogout} />}
      {withSidebar ? (
        <div className="app-layout">
          <Sidebar screen={screen} setScreen={setScreen} />
          <div className="app-content">{renderScreen()}</div>
        </div>
      ) : renderScreen()}
    </>
  );
}
